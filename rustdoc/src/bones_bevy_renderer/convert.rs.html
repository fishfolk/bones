8pub trait IntoBevy<T> {
9    fn into_bevy(self) -> T;
10}
11pub trait IntoBones<T> {
12    fn into_bones(self) -> T;
13}

15impl IntoBevy<Viewport> for bones::Viewport {
16    fn into_bevy(self) -> Viewport {
17        Viewport {
18            physical_position: self.position,
19            physical_size: self.size,
20            depth: self.depth_min..self.depth_max,
21        }
22    }
23}
24
25impl IntoBevy<Color> for bones::Color {
26    fn into_bevy(self) -> Color {
27        Color::Rgba {
28            red: self.r(),
29            green: self.g(),
30            blue: self.b(),
31            alpha: self.a(),
32        }
33    }
34}

36impl IntoBevy<Transform> for bones::Transform {
37    fn into_bevy(self) -> Transform {
38        Transform {
39            translation: self.translation,
40            rotation: self.rotation,
41            scale: self.scale,
42        }
43    }
44}
