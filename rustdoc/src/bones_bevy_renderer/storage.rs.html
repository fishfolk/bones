22    impl StorageApi for StorageBackend {
23        fn save(&mut self, data: Vec<SchemaBox>) {
24            let mut buffer = Vec::new();
25            let mut serializer = serde_yaml::Serializer::new(&mut buffer);
26            LoadedStorage(data)
27                .serialize(&mut serializer)
28                .expect("Failed to serialize to storage file.");
29            let data = String::from_utf8(buffer).unwrap();
30            let window = web_sys::window().unwrap();
31            let storage = window.local_storage().unwrap().unwrap();
32            storage.set_item(&self.storage_key, &data).unwrap();
33        }
34
35        fn load(&mut self) -> Vec<SchemaBox> {
36            let window = web_sys::window().unwrap();
37            let storage = window.local_storage().unwrap().unwrap();
38            let Some(data) = storage.get_item(&self.storage_key).unwrap() else {
39                return default();
40            };
41
42            let Ok(loaded) = serde_yaml::from_str::<LoadedStorage>(&data) else {
43                return default();
44            };
45            loaded.0
46        }
47    }
48}

60    impl StorageBackend {
61        pub fn new(qualifier: &str, organization: &str, application: &str) -> Self {
62            let project_dirs = directories::ProjectDirs::from(qualifier, organization, application)
63                .expect("Identify system data dir path");
64            Self {
65                storage_path: project_dirs.data_dir().join("storage.yml"),
66            }
67        }
68    }
69
70    impl StorageApi for StorageBackend {
71        fn save(&mut self, data: Vec<SchemaBox>) {
72            let file = std::fs::OpenOptions::new()
73                .write(true)
74                .truncate(true)
75                .create(true)
76                .open(&self.storage_path)
77                .expect("Failed to open storage file");
78            let mut serializer = serde_yaml::Serializer::new(file);
79            LoadedStorage(data)
80                .serialize(&mut serializer)
81                .expect("Failed to serialize to storage file.");
82        }
83
84        fn load(&mut self) -> Vec<SchemaBox> {
85            if self.storage_path.exists() {
86                let result: anyhow::Result<LoadedStorage> = (|| {
87                    let file = std::fs::OpenOptions::new()
88                        .read(true)
89                        .open(&self.storage_path)
90                        .context("Failed to open storage file")?;
91                    let loaded: LoadedStorage = serde_yaml::from_reader(file)
92                        .context("Failed to deserialize storage file")?;
93
94                    anyhow::Result::Ok(loaded)
95                })();
96                match result {
97                    Ok(loaded) => loaded.0,
98                    Err(e) => {
99                        error!(
100                            "Error deserializing storage file, ignoring file, \
101                        data will be overwritten when saved: {e:?}"
102                        );
103                        default()
104                    }
105                }
106            } else {
107                std::fs::create_dir_all(self.storage_path.parent().unwrap()).unwrap();
108                default()
109            }
110        }
111    }

114struct LoadedStorage(Vec<SchemaBox>);
115impl Serialize for LoadedStorage {
116    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
117    where
118        S: serde::Serializer,
119    {
120        let data: HashMap<String, SchemaRef> = self
121            .0
122            .iter()
123            .map(|x| (x.schema().full_name.to_string(), x.as_ref()))
124            .collect();
125
126        use serde::ser::SerializeMap;
127        let mut map = serializer.serialize_map(Some(data.len()))?;
128
129        for (key, value) in data {
130            map.serialize_key(&key)?;
131            map.serialize_value(&SchemaSerializer(value))?;
132        }
133
134        map.end()
135    }
136}
137impl<'de> Deserialize<'de> for LoadedStorage {
138    fn deserialize<D>(deserializer: D) -> Result<Self, D::Error>
139    where
140        D: serde::Deserializer<'de>,
141    {
142        deserializer.deserialize_map(LoadedStorageVisitor).map(Self)
143    }
144}
145struct LoadedStorageVisitor;
146impl<'de> Visitor<'de> for LoadedStorageVisitor {
147    type Value = Vec<SchemaBox>;
148    fn expecting(&self, formatter: &mut std::fmt::Formatter) -> std::fmt::Result {
149        write!(formatter, "Mapping of string type names to type data.")
150    }
151    fn visit_map<A>(self, mut map: A) -> Result<Self::Value, A::Error>
152    where
153        A: serde::de::MapAccess<'de>,
154    {
155        let mut data = Vec::new();
156        while let Some(type_name) = map.next_key::<String>()? {
157            let Some(schema) = SCHEMA_REGISTRY
158                .schemas
159                .iter()
160                .find(|schema| schema.full_name.as_ref() == type_name)
161            else {
162                error!(
163                    "\n\nCannot find schema registration for `{}` while loading persisted \
164                        storage. This means you that you need to call \
165                        `{}::schema()` to register your persisted storage type before \
166                        creating the `BonesBevyRenderer` or that there is data from an old \
167                        version of the app inside of the persistent storage file.\n\n",
168                    type_name, type_name,
169                );
170                continue;
171            };
172
173            data.push(map.next_value_seed(SchemaDeserializer(schema))?);
174        }
175
176        Ok(data)
177    }
