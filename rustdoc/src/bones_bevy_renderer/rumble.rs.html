10/// Handles rumble requests from the Bones framework and translates them to Bevy rumble requests
11pub fn handle_bones_rumble(
12    game: ResMut<BonesGame>,
13    mut rumble_requests: EventWriter<BevyGamepadRumbleRequest>,
14) {
15    if let Some(mut bones_rumble_requests) = game.get_shared_resource_mut::<bones::GamepadsRumble>()
16    {
17        while let Some(request) = bones_rumble_requests.requests.pop_front() {
18            match request {
19                bones::GamepadRumbleRequest::AddRumble {
20                    gamepad,
21                    intensity,
22                    duration,
23                } => {
24                    send_rumble_request(&gamepad, &intensity, &duration, &mut rumble_requests);
25                }
26                bones::GamepadRumbleRequest::SetRumble {
27                    gamepad,
28                    intensity,
29                    duration,
30                } => {
31                    // First, stop the current rumble
32                    let stop_gamepad = Gamepad::new(gamepad as usize);
33                    rumble_requests.send(BevyGamepadRumbleRequest::Stop {
34                        gamepad: stop_gamepad,
35                    });
36
37                    // Then, add the new rumble
38                    send_rumble_request(&gamepad, &intensity, &duration, &mut rumble_requests);
39                }
40                bones::GamepadRumbleRequest::Stop { gamepad } => {
41                    let gamepad = Gamepad::new(gamepad as usize);
42                    rumble_requests.send(BevyGamepadRumbleRequest::Stop { gamepad });
43                }
44            }
45        }
46    }
47}
48
49/// Helper function to send a rumble request to Bevy
50fn send_rumble_request(
51    gamepad: &u32,
52    intensity: &bones::GamepadRumbleIntensity,
53    duration: &f32,
54    rumble_requests: &mut EventWriter<BevyGamepadRumbleRequest>,
55) {
56    let bevy_intensity = BevyGamepadRumbleIntensity {
57        strong_motor: intensity.strong_motor(),
58        weak_motor: intensity.weak_motor(),
59    };
60
61    let gamepad = Gamepad::new(*gamepad as usize);
62
63    rumble_requests.send(BevyGamepadRumbleRequest::Add {
64        gamepad,
65        intensity: bevy_intensity,
66        duration: Duration::from_secs_f32(*duration),
67    });
68}
