26pub fn get_bones_input(
27    mut mouse_button_input_events: EventReader<MouseButtonInput>,
28    mut mouse_motion_events: EventReader<MouseMotion>,
29    mut mouse_wheel_events: EventReader<MouseWheel>,
30    mut keyboard_events: EventReader<KeyboardInput>,
31) -> (
32    bones::MouseInputs,
33    bones::KeyboardInputs,
34    bones::GamepadInputs,
35) {
36    // TODO: investigate possible ways to avoid allocating vectors every frame for event lists.
37    (
38        bones::MouseInputs {
39            movement: mouse_motion_events
40                .iter()
41                .last()
42                .map(|x| x.delta)
43                .unwrap_or_default(),
44            wheel_events: mouse_wheel_events
45                .iter()
46                .map(|event| bones::MouseScrollEvent {
47                    unit: event.unit.into_bones(),
48                    movement: Vec2::new(event.x, event.y),
49                })
50                .collect(),
51            button_events: mouse_button_input_events
52                .iter()
53                .map(|event| bones::MouseButtonEvent {
54                    button: event.button.into_bones(),
55                    state: event.state.into_bones(),
56                })
57                .collect(),
58        },
59        bones::KeyboardInputs {
60            key_events: keyboard_events
61                .iter()
62                .map(|event| bones::KeyboardEvent {
63                    scan_code: event.scan_code,
64                    key_code: event.key_code.map(|x| x.into_bones()).into(),
65                    button_state: event.state.into_bones(),
66                })
67                .collect(),
68        },
69        process_gamepad_events(),
70    )
71}

81// Source: https://bevy-cheatbook.github.io/cookbook/cursor2world.html
82pub fn get_mouse_position(
83    mut q_primary_windows: Query<&Window, With<PrimaryWindow>>,
84    q_camera: Query<(&Camera, &GlobalTransform)>,
85) -> (Option<Vec2>, Option<Vec2>) {
86    match q_primary_windows
87        .get_single_mut()
88        .ok()
89        .and_then(Window::cursor_position)
90    {
91        None => (None, None),
92        screen_pos @ Some(sp) => match q_camera.get_single() {
93            Err(_) => (screen_pos, None),
94            Ok((camera, camera_transform)) => {
95                let world_pos = camera.viewport_to_world_2d(camera_transform, sp);
96                (screen_pos, world_pos)
97            }
98        },
99    }
