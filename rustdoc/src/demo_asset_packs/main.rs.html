1use bones_bevy_renderer::BonesBevyRenderer;
2use bones_framework::prelude::*;
3
4//
5// NOTE: You must run this example from within the `demos/asset_packs` folder. Also, be sure to
6// look at the `assets/` and `packs/` folders to see the asset files for this example.
7//
8
9/// Our "core" asset type.
10#[derive(HasSchema, Clone, Default)]
11#[repr(C)]
12// We must mark this as a metadata asset, and we set the type to "game".
13//
14// This means that any files with names like `game.yaml`, `game.yml`, `game.json`, `name.game.yaml`,
15// etc. will be loaded as a `GameMeta` asset.
16#[type_data(metadata_asset("game"))]
17struct GameMeta {
18    title: String,
19    core_items: SVec<i32>,
20}
21
22/// Our "supplementary" asset type.
23#[derive(HasSchema, Clone, Default)]
24#[repr(C)]
25#[type_data(metadata_asset("data"))]
26struct PackMeta {
27    items: SVec<i32>,
28}
29
30fn main() {
31    // Setup logging
32    setup_logs!();
33
34    // First create bones game.
35    let mut game = Game::new();
36
37    game
38        // We initialize the asset server.
39        .init_shared_resource::<AssetServer>();
40
41    // We must register all of our asset types before they can be loaded by the asset server. This
42    // may be done by calling schema() on each of our types, to register them with the schema
43    // registry.
44    GameMeta::register_schema();
45    PackMeta::register_schema();
46
47    // Create a new session for the game menu. Each session is it's own bones world with it's own
48    // plugins, systems, and entities.
49    game.sessions
50        .create_with("menu", |builder: &mut SessionBuilder| {
51            // Install the default bones_framework plugin for this session
52            builder
53                .install_plugin(DefaultSessionPlugin)
54                // Add our menu system to the update stage
55                .add_system_to_stage(Update, menu_system);
56        });
57
58    BonesBevyRenderer::new(game).app().run();
59}
60
61/// System to render the home menu.
62fn menu_system(
63    egui_ctx: Res<EguiCtx>,
64    core_meta: Root<GameMeta>,
65    all_packs: AllPacksData<GameMeta, PackMeta>,
66) {
67    egui::CentralPanel::default()
68        .frame(egui::Frame::none().outer_margin(egui::Margin::same(32.0)))
69        .show(&egui_ctx, |ui| {
70            // Use the title that has been loaded from the asset
71            ui.heading(&core_meta.title);
72
73            ui.separator();
74
75            ui.label(egui::RichText::new("Items from all asset packs:"));
76
77            // Show the numbers from all of the asset packs
78            egui::Grid::new("pack-items").num_columns(1).show(ui, |ui| {
79                for item in all_packs.iter_with(
80                    |core| core.core_items.iter().copied(),
81                    |pack| pack.items.iter().copied(),
82                ) {
83                    ui.label(item.to_string());
84                    ui.end_row();
85                }
86            });
87        });
88}
